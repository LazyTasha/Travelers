CREATE TABLE gg_region
(
region_id number(2,0) NOT NULL,
region_name varchar2(40) NOT NULL UNIQUE,
continent varchar2(20),
PRIMARY KEY (region_id)
);

CREATE TABLE gg_country
(
country_id number(4,0) NOT NULL,
country_name varchar2(60) ,
country_code varchar(10) NOT NULL UNIQUE,
region_id number(2,0) NOT NULL,
PRIMARY KEY (country_id),
foreign key(region_id) references gg_region(region_id)
);

CREATE TABLE gg_coordinate
(
country_id number(4,0) NOT NULL,
coord_long number(10,6) NOT NULL,
coord_lat number(10,6) NOT NULL,
coord_order number(3),
coord_id number(10,0) NOT NULL,
PRIMARY KEY (coord_id),
foreign key(country_id) references gg_country(country_id)
);

CREATE SEQUENCE SEQ_gg_coordinate_coord_id INCREMENT BY 1 START WITH 1;


CREATE TABLE gg_user
(
user_id varchar2(20) NOT NULL,
passwd varchar2(30) NOT NULL,
email varchar2(50) NOT NULL UNIQUE,
gender number(2,0),
reg_date date DEFAULT SYSDATE NOT NULL,
user_name varchar2(30) NOT NULL,
user_level number(2,0) NOT NULL,
PRIMARY KEY (user_id)
);

CREATE TABLE gg_trip_board
(
tb_no number(10,0) NOT NULL,
tb_title varchar2(100) NOT NULL,
user_id varchar2(20),
tb_content varchar2(4000) NOT NULL,
tb_reg_date date DEFAULT SYSDATE NOT NULL,
tb_v_count number(10,0) DEFAULT 0 NOT NULL,
tb_m_num number(10,0) DEFAULT 1 NOT NULL,
tb_notice number(2,0),
tb_talk varchar2(50),
PRIMARY KEY (tb_no),
foreign key(user_id) references gg_user(user_id)
on delete set null
);

CREATE SEQUENCE SEQ_gg_trip_board_tb_no INCREMENT BY 1 START WITH 1;


CREATE TABLE gg_trip_detail
(
td_trip_id number(4,0) NOT NULL,
tb_no number(10,0) NOT NULL,
primary key(td_trip_id),
foreign key(tb_no) references gg_trip_board(tb_no)
on delete cascade
);

CREATE SEQUENCE SEQ_gg_trip_detail_td_trip_id INCREMENT BY 1 START WITH 1;

CREATE TABLE gg_calendar
(
cal_start_date date NOT NULL,
cal_end_date date NOT NULL,
coord_id number(10,0) NOT NULL,
td_trip_id number(4,0) NOT NULL,
foreign key(td_trip_id) references gg_trip_detail(td_trip_id)
on delete cascade,
foreign key(coord_id) references gg_coordinate(coord_id)
);


CREATE TABLE gg_member
(
user_id varchar2(20) NOT NULL,
td_trip_id number(4,0) NOT NULL,
foreign key(user_id) references gg_user(user_id)
on delete cascade,
foreign key(td_trip_id) references gg_trip_detail(td_trip_id)
on delete cascade
);


CREATE TABLE gg_comment
(
c_id number(10,0) DEFAULT 1 NOT NULL,
tb_no number(10,0) NOT NULL,
user_id varchar2(20),
c_content varchar2(500) NOT NULL,
c_reg_date date NOT NULL,
PRIMARY KEY (c_id),
foreign key(tb_no) references gg_trip_board(tb_no)
on delete cascade,
foreign key(user_id) references gg_user(user_id)
on delete set null
);

CREATE SEQUENCE SEQ_gg_comment_comment_id INCREMENT BY 1 START WITH 1;


CREATE TABLE gg_tag
(
tag_id number(4,0) NOT NULL,
tag_value varchar2(100) NOT NULL UNIQUE,
PRIMARY KEY (tag_id)
);

CREATE SEQUENCE SEQ_gg_tag_tag_id INCREMENT BY 1 START WITH 1;


CREATE TABLE gg_trip_tag
(
tb_no number(10,0) NOT NULL,
tag_id number(4,0) NOT NULL,
foreign key(tb_no) references gg_trip_board
on delete cascade,
foreign key(tag_id) references gg_tag(tag_id)
on delete cascade
);


CREATE TABLE gg_user_tag
(
user_id varchar2(20) NOT NULL,
tag_id number(4,0) DEFAULT 1 NOT NULL,
foreign key(user_id) references gg_user(user_id)
on delete cascade,
foreign key(tag_id) references gg_tag(tag_id)
on delete cascade
);

(
photo_id number(7,0) NOT NULL,
photo_url varchar2(2048) NOT NULL,
alb_reg_date date NOT NULL,
tb_no number(10,0) NOT NULL,
PRIMARY KEY (photo_id),
foreign key(tb_no) references gg_trip_board(tb_no)
on delete cascade
);
